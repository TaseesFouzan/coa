ShiftRegister74HC595 sr(2,12,10,11);

B00000110,
B01011011,
B01001111,
B01100110,
B01101101,
B01111101,
B00000111,

B01101111
};

byte rowPins[numCols]= {6,7,8,9};

{
Serial.begin(9600);
}
void loop()
{
char keypressed = myKeypad.getKey();
Serial.println("keypressed : "+String(keypressed));

{
Serial.print(keypressed);
